# 기본유형

Primitives

```typescript
// Primitives

// number
let age : number ;

age = 12;

// string
let userName = string;

userName : "max"

// boolean

let isInstructor : boolean

isInstructor = true

let hobbies : null
hobbies = 12 // error
```

More complex types

```typescript

// 배열은 string[] , number[], boolean[] 등으로 표현 가능

let hobbies : string[];
// 문자 배열을 가진다.

hobbies = ['Sports', 'Cooking'] // success

hobbies = ['Sports', 'Cooking', 12] // error

// 타입 할당
let person : {
  name : string,
  age : number
};

// success
person = {
  name : "Max",
  age: 12
}

// error
person = {
  isEmployee : true
}

// 객체 배열
let people : {
  name : string,
  age:numer
}[]
```
